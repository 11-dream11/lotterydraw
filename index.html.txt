        function updateTimer() {
            let h = String(Math.floor(timeLeft / 3600)).padStart(2, '0');

            let s = String(timeLeft % 60).padStart(2, '0');
            document.getElementById('timer').textContent = `${h}:${m}:${s}`;

                const win = Math.floor(100000 + Math.random() * 900000);
                document.getElementById('winning-number').textContent = win;
                document.getElementById('draw-time').textContent = new Date().toLocaleString('en-IN');

                timeLeft = 18000;
            } else timeLeft--;
        }
